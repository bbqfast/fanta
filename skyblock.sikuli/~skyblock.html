def getTrees():
    cnt=1
    rocks=[imgTree1,imgTree2,imgTree3,imgTree4,imgTree5,imgTree6,imgTree7,imgTree8,imgTree9]
    while True:
        for r in rocks:
            r=r.similar(treeSimilar)
            lib.log("----------------- " + str(cnt))
            if lib.logExists(game, r):
                cnt=cnt+1
                lib.myClick(game, r, holdDur=3.5)
